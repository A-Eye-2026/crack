# [17강] Convolutional Neural Networks (CNN) 핵심 코드
# Fashion MNIST 데이터셋을 활용한 이미지 분류 모델 구축 예제입니다.

import tensorflow as tf
from tensorflow import keras
from tensorflow.keras import layers
import numpy as np

# 1. 데이터 로드 및 전처리
(train_input, train_target), (test_input, test_target) = keras.datasets.fashion_mnist.load_data()

# 흑백 이미지이므로 채널 차원(1)을 추가하고 0~1 사이로 정규화합니다.
train_scaled = train_input.reshape(-1, 28, 28, 1) / 255.0
test_scaled = test_input.reshape(-1, 28, 28, 1) / 255.0

# 2. CNN 모델 구축
model = keras.Sequential()

# 첫 번째 합성곱 층: 32개 필터, (3,3) 커널, ReLU 활성화, Same 패딩
model.add(layers.Conv2D(32, kernel_size=3, activation='relu', padding='same', input_shape=(28,28,1)))
# 풀링 층: (2,2) 크기로 특성 맵 축소
model.add(layers.MaxPooling2D(2))

# 두 번째 합성곱 층: 64개 필터, (3,3) 커널, ReLU 활성화, Same 패딩
model.add(layers.Conv2D(64, kernel_size=3, activation='relu', padding='same'))
model.add(layers.MaxPooling2D(2))

# 3. 출력층 구성
model.add(layers.Flatten())      # 1차원으로 펼치기
model.add(layers.Dense(100, activation='relu'))  # 밀집층
model.add(layers.Dropout(0.4))   # 과대적합 방지
model.add(layers.Dense(10, activation='softmax')) # 10개 클래스 확률 출력

# 4. 모델 컴파일 및 훈련
model.compile(optimizer='adam', 
              loss='sparse_categorical_crossentropy', 
              metrics=['accuracy'])

# 체크포인트 및 조기 종료 콜백 설정
checkpoint_cb = keras.callbacks.ModelCheckpoint('best-cnn-model.h5', save_best_only=True)
early_stopping_cb = keras.callbacks.EarlyStopping(patience=2, restore_best_weights=True)

history = model.fit(train_scaled, train_target, epochs=20,
                    validation_split=0.2, 
                    callbacks=[checkpoint_cb, early_stopping_cb])

# 5. 모델 평가
test_loss, test_acc = model.evaluate(test_scaled, test_target)
print(f"테스트 정확도: {test_acc:.4f}")
